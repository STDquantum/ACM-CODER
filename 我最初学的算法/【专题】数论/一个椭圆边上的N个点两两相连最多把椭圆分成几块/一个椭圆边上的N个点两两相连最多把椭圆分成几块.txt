		for (i = 0; i <= n-2; i++)

			bufv += i*(n-2-i);
			bufe += i*(n-2-i)+1;
		}
		v = n+(n*bufv)/4;//²»ÒªÐ´³Év = n+(n/4)*bufv;

}
