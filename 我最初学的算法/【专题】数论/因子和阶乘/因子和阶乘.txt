#include <stdio.h>
#define scanf scanf_s

bool is_prime (int);

int main ()
{
	int prime[1254], i, j = 0, n;
	for (i = 2; i <= 10000;  i++)//生成素数表
	{
		if (is_prime(i))
			prime[j++] = i;
	}

	while (EOF != scanf ("%d%*c", &n))
	{
		int maxp = -1, p[1254]={0}, k;
		for (i = 1; i <= n; i++)
		{
			int buf = i;
			for (k = 0; k < j; k++)
			{
				while (0 == buf % prime[k])
				{
					buf /= prime[k];
					p[k]++;
					if (k > maxp)
						maxp = k;
				}
			}
		}
		//输出
		for (i = 0; i <= maxp; i++)
		{
			if (p[i] != 0)
				printf ("%d ", p[i]);
		}
		printf ("\n");
	}

	return 0;
}

bool is_prime (int a)
{
	for (int i = 2;  i*i <= a; i++)
		if (0 == a%i)
			return false;
	return true;
}
