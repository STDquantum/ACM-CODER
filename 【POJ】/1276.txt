/*这里的价值就是重量*/
#include <stdio.h>
#include <string.h>
#define MAXV 100010
#define MAXN 150 //注意是∑log2(n) 10(N)*log2(n)
int w[MAXN], d[MAXV], v[MAXN];

int max(int a, int b){return a>b?a:b;}
int main ()
{
#ifdef SHY
	freopen("e:\\1.txt","r",stdin);
#endif
	int c, n;
	while(~scanf ("%d %d%*c", &c, &n))
	{
		int len = 0, num, val;
		//二进制拆分成能组合成所有1~num的数量
		for (int i = 0; i < n; i++)
		{
			scanf ("%d %d%*c", &num, &val);
			for(int j = 1; j <= num; num -= j, j <<= 1)
			{
				w[len] = j*val;
				if (w[len] > c)//把大于背包容量的结果去掉
					break;
				v[len++] = j*val;
			}
			if (num > 0)
			{
				w[len] = num*val;
				v[len++] = num*val;
			}
		}
		//0-1背包
		memset(d,0,sizeof(d));
		for (int i = 0; i < len; i++)
		{
			for (int j = c; j >= w[i]; j--)
			{
				d[j] = max(d[j],d[j-w[i]]+v[i]);
			}
		}
		printf ("%d\n", d[c]);
	}
	return 0;
}
