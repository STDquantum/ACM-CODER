struct NODE{
	NODE* ch[26];
	int sum;
	NODE(){
		sum = 0;
		clc(ch,0);
	}
};

void insert(NODE *rt, char *s)//²åÈës
{
	int n = strlen(s);
	rep(i,0,n)
	{
		int c = s[i]-'a';
		if(!rt->ch[c])
			rt->ch[c] = new NODE();
		rt = rt->ch[c];
		rt->sum++;
	}
}
int query(NODE *rt, char *s)
{
	int n = strlen(s);
	rep(i,0,n)
	{
		int c = s[i]-'a';
		if(!rt->ch[c]) return 0;
		rt = rt->ch[c];
	}
	return rt->sum;
}
