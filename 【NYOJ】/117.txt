#include <stdio.h>
const int MAXN = 1000011; 
int a[MAXN], t[MAXN], n;
long long ans;

void merge_sort(int, int);
int main ()
{
#ifdef DSHY
	freopen("e:\\1.txt","r",stdin);
#endif
	int t;
	scanf ("%d%*c", &t);
	while(t--)
	{
		scanf ("%d%*c", &n);
		for (int i = 0; i < n; i++)
			scanf ("%d%*c", &a[i]);
		ans = 0;
		merge_sort(0,n);
		printf ("%lld\n", ans);
	}
	return 0;
}

void merge_sort(int l, int r)
{
	if (r-l <= 1)
		return;
	int m = l+(r-l)/2;
	merge_sort(l, m);//ÅÅÐò×ó°ë±ß
	merge_sort(m, r);//ÅÅÐòÓÒ°ë±ß
	int p = l, q = m, i = l;
	while(p < m || q < r)
	{
		if (q >= r || (p < m && a[p] <= a[q]))
			t[i++] = a[p++];
		else
		{
			t[i++] = a[q++];
			ans += m-p;
		}
	}
	for (i = l; i < r; i++)
		a[i] = t[i];
}
