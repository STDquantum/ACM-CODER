#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define MAXN 25
int a[MAXN], dp[MAXN];

int main()
{
#ifdef SHY
	freopen("e:\\1.txt","r",stdin);
#endif
	int t;
	scanf("%d%*c", &t);
	while(t--)
	{
		int n, ans = 1;
		scanf("%d%*c", &n);
		for (int i = 0; i < n; i++)
			scanf("%d%*c", &a[i]);
		for (int i = 0; i < n; i++)
		{
			dp[i] = 1;
			for (int j = 0; j < i; j++)
			{
				if (a[j] > a[i])
					dp[i] = max(dp[i], dp[j]+1);
			}
			ans = max(ans,dp[i]);
		}
		printf("%d\n", ans);
	}
	return 0;
}
