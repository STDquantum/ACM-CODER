#include <cstring>
#include <algorithm>
using namespace std;
#define MAXN 110
int a[MAXN][MAXN], dp[MAXN][MAXN];

int main ()
{
    #ifdef SHY
        freopen("e:\\1.txt", "r", stdin);
    #endif
    int t;
    scanf ("%d%*c", &t);
    while(t--)
    {
        int n;
        scanf ("%d%*c", &n);
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j <= i; j++)
            {
                scanf ("%d%*c", &a[i][j]);
                dp[i][j] = a[i][j];
            }
        }
        for (int i = n-2; i >= 0; i --)
            for (int j = 0; j <= i; j++)
                dp[i][j] = max(dp[i+1][j], dp[i+1][j+1])+a[i][j];
        printf ("%d\n", dp[0][0]);
    }
    return 0;
}

int sloved(int i, int j)
{
    if (d[i][j] >= 0)
        return d[i][j];
    if (i == n-1)
        return d[i][j] = a[i][j]+0;
