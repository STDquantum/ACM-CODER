#include <stdio.h>

__int64 MM(int a)
{
 if(a==1){return 0;}
 else if(a==2){return 1;}
 else return (a-1)*(MM(a-1)+MM(a-2));
}//Çó´íÅÅÊı

int main ()
{
	int t;
	scanf ("%d%*c", &t);
	while (t--)
	{
		int n, m, i, j;
		__int64 sum = 1,s = 1;
		scanf ("%d %d%*c", &n, &m);
		for (i = 0; i < m; i++, n--)
		{
			sum *= n;
			s *= m-i;
		}
		printf ("%I64d\n", sum/s*MM(m));
	}

	return 0;
}
