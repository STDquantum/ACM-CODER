LL n,MOD;

LL mul_mod(LL a, LL b)//a*b¶ÔMODÈ¡Óà
{
	a %= MOD;
	LL ans = 0;
	while(b)
	{
		if(b&1)
		{
			ans += a;
			if(ans >= MOD) ans -= MOD;
		}
		a <<= 1;
		if(a >= MOD) a -= MOD;
		b >>= 1;
	}
	return ans;
}
LL pow_mod(LL x, LL n)
{
    LL ans = 1;
    while(n)
    {
        if(n&1) ans = mul_mod(ans,x);
        x = mul_mod(x,x);
        n >>= 1;
    }
    return ans;
}
