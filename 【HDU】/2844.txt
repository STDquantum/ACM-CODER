{

                for (int i = 0; i < n; i++)
			scanf ("%d%*c", &v[i]);

		{
                        scanf ("%d%*c", &num);
                        if (num*v[i] >= c)
			{
                                for (int j = v[i]; j <= c; j++)
					d[j] = max(d[j],d[j-v[i]]+v[i]);

                        else
			{
                                for (int j = 1; j <= num; num -= j, j <<= 1)
				{
					int buf = j*v[i];
                                        for (int k = c; k >= buf; k--)
						d[k] = max(d[k],d[k-buf]+buf);

				}

		}
